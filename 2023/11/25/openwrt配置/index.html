    命令修改端口映射

    vim /etc/config/firewall
config redirect                                 
        option dest 'lan'                       
        option target 'DNAT'                    
        option name 'ssh'                       
        option src 'wan'                        
        option src_dport '33333'                
        option dest_ip '192.168.100.1'          
        option dest_port '80'

/etc/init.d/firewall reload

    接口配置文件位置

    vim /etc/config/network

    重启网络

    /etc/init.d/network restart

    定时脚本，避免拨号掉线没有重拨

    #!/bin/bash
HOST="www.baidu.com"

for i in $(seq 1 10)
do
ping -c 3 $HOST > /dev/null
if [ $? -eq 0 ]
then
    break
else
    ifconfig eth0 down
    ifconfig eth0 up
    sleep 30
fi
done
