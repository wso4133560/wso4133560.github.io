    包安装

    conda install Pillow==9.4.0
    pip install apache-tvm Pillow

    github下载官方源码

    git clone https://github.com/apache/tvm.git -b v0.14.0
    git submodule update –init –recursive

    安装需要的库

    sudo apt-get install libgtest-dev libgmock-dev verilator pypy3-dev
    pip install pytest numpy decorator scipy psutil typing_extensions attrs

    编译步骤

    export TVM_PATH=
    export VTA_HW_PATH=$TVM_PATH/3rdparty/vta-hw

    修改cmake/config.cmake
    set(USE_VTA_FSIM ON)

    mkdir build
    cp cmake/config.cmake build/.
    cd build && cmake .. && make -j4

    export PYTHONPATH=/path/to/vta/python:${PYTHONPATH}
    export PYTHONPATH=/path/to/tvm/python:${PYTHONPATH}

    python /vta/tests/python/integration/test_benchmark_topi_conv2d.py
